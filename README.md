Import RPi as GPIO
import time

ledPin = 11

def setup():
       GPIO.setmode(GPIO.BOARD)
       GPIO.setup(ledPin, GPIO.OUT)
       GPIO.output(ledPin, GPIO.LOW)

              GPIO.output(ledPin, GPIO.HIGH)
               print (‘Led Turned On >>>’)
               time.sleep(1)
               GPIO.output(ledPin, GPIO.LOW)
               print (‘Led Turned Off <<<’)
               time.sleep(1)

def destroy():
       GPIO.cleanup()
